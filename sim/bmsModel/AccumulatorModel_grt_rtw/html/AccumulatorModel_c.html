533        f1, nx,



534        1, pivots,



535        1);

537        Delta, nx,



538        1, 0);



539



540      /* ytmp = ytmp + delta



541         ztmp = (ytmp - yn)/h



542       */



543      for (i = 0; i < nx; i++) {



544        x1[i] += Delta[i];



545        ztmp[i] = (x1[i] - x0[i])/hN;



546      }



547    }

550       ttmp = t0;

552       ttmp = ttmp + hN



553       ytmp0 = ytmp;



554       for iter = 1:NewtIter



555       rhs = (ytmp0 - ytmp) + hN*feval(odefun,ttmp,ytmp,extraArgs{:});



556       Delta = ( U \ ( L \ rhs ) );



557       ytmp = ytmp + Delta;



558       end



559       end

561    for (k = 1; k < N[j]; k++) {



562      t1 = t0 + k*hN;



563      (void) memcpy(x1start, x1,



564                    (uint_T)nx*sizeof(real_T));



565      rtsiSetT(si, t1);



566      rtsiSetdX(si, f1);

568        AccumulatorModel_step();



569        AccumulatorModel_forcingfunction();



570        if (iter == 0) {

573          }



574        } else {



575          for (i = 0; i < nx; i++) {



576            Delta[i] = hN*f1[i];



577            xtmp[i] = (x1[i]-x1start[i]);



578          }



579



580          /* rhs = hN*f(tn,ytmp) - M*(ytmp-yn) */



581          for (col = 0; col < nx; col++) {



582            for (rowidx = Mpattern_jc[col]; rowidx < Mpattern_jc[col+1]; rowidx



583                 ++) {

592        AccumulatorModel_step();



593        local_evaluateMassMatrix(si,M1 );



594        for (i = 0; i < rtsiGetSolverMassMatrixNzMax(si); i++) {



595          M1[i] -= M[i];



596        }



597



598        /* rhs = rhs - Mdel*ztmp*h */



599        for (col = 0; col < nx; col++) {



600          for (rowidx = Mpattern_jc[col]; rowidx < Mpattern_jc[col+1]; rowidx++)



601          {



602            real_T m_row_col = M1[rowidx];



603            row = Mpattern_ir[rowidx];



604            Delta[row] -= hN*m_row_col*ztmp[col];



605          }



606        }

608        rt_ForwardSubstitutionRR_Dbl(W, Delta,



609          f1, nx,



610          1, pivots,



611          1);



612        rt_BackwardSubstitutionRR_Dbl(W+nx*nx-1, f1+nx-1,



613          Delta, nx,



614          1, 0);

616        /* ytmp = ytmp + Delta



617           ztmp = (ytmp - ytmp0)/h



618         */



619        for (i = 0; i < nx; i++) {



620          x1[i] += Delta[i];



621          ztmp[i] = (x1[i] - x1start[i])/hN;



622        }



623      }



624    }



625



626    /* Extrapolate to order j



627       E(:,j) = ytmp



628       for k = j:-1:2



629       coef = N(k-1)/(N(j) - N(k-1))



630       E(:,k-1) = E(:,k) + coef*( E(:,k) - E(:,k-1) )



631       end



632     */



633    (void) memcpy(&(E[nx*j]), x1,



634                  (uint_T)nx*sizeof(real_T));



635    for (k = j; k > 0; k--) {

637      for (i = 0; i < nx; i++) {

639      }



640



641      (void) memcpy(&(E[nx*(k-1)]), x1,



642                    (uint_T)nx*sizeof(real_T));



643    }



644

648    }



649



650    (void) memcpy(&(Edot[nx*j]), xdot,



651                  (uint_T)nx*sizeof(real_T));



652    for (k = j; k > 0; k--) {



653      real_T coef = (real_T)(N[k-1]) / (N[j]-N[k-1]);



654      for (i = 0; i < nx; i++) {



655        xdot[i] = Edot[nx*k+i] + coef*(Edot[nx*k+i] - Edot[nx*(k-1)+i]);



656      }



657



658      (void) memcpy(&(Edot[nx*(k-1)]), xdot,



659                    nx*sizeof(real_T));



660    }



661  }



662



663  /* x1 = E(:,1); */



664  (void) memcpy(x1, E,



665                (uint_T)nx*sizeof(real_T));



666



667  /* Extrapolated xdot */



668  (void) memcpy(xdot, Edot,



669                nx*sizeof(real_T));



670



671  /* t1 = t0 + h; */



672  rtsiSetT(si,rtsiGetSolverStopTime(si));



673  rtsiSetSimTimeStep(si,MAJOR_TIME_STEP);



674}
