546         ztmp = (ytmp - yn)/h

548      for (i = 0; i < nx; i++) {

555       ttmp = t0;

558       ytmp0 = ytmp;



559       for iter = 1:NewtIter



560       rhs = (ytmp0 - ytmp) + hN*feval(odefun,ttmp,ytmp,extraArgs{:});



561       Delta = ( U \ ( L \ rhs ) );



562       ytmp = ytmp + Delta;



563       end



564       end



565     */



566    for (k = 1; k < N[j]; k++) {



567      t1 = t0 + k*hN;



568      (void) memcpy(x1start, x1,



569                    (uint_T)nx*sizeof(real_T));



570      rtsiSetT(si, t1);



571      rtsiSetdX(si, f1);



572      for (iter = 0; iter < numIter; iter++) {



573        AccumulatorModel_step();



574        AccumulatorModel_forcingfunction();



575        if (iter == 0) {



576          for (i = 0; i < nx; i++) {



577            Delta[i] = hN*f1[i];



578          }



579        } else {



580          for (i = 0; i < nx; i++) {



581            Delta[i] = hN*f1[i];



582            xtmp[i] = (x1[i]-x1start[i]);



583          }

587            for (rowidx = Mpattern_jc[col]; rowidx < Mpattern_jc[col+1]; rowidx



588                 ++) {

597        AccumulatorModel_step();

600          M1[i] -= M[i];



601        }

604        for (col = 0; col < nx; col++) {



605          for (rowidx = Mpattern_jc[col]; rowidx < Mpattern_jc[col+1]; rowidx++)



606          {

609            Delta[row] -= hN*m_row_col*ztmp[col];



610          }



611        }

613        rt_ForwardSubstitutionRR_Dbl(W, Delta,



614          f1, nx,



615          1, pivots,



616          1);



617        rt_BackwardSubstitutionRR_Dbl(W+nx*nx-1, f1+nx-1,



618          Delta, nx,



619          1, 0);

622           ztmp = (ytmp - ytmp0)/h



623         */



624        for (i = 0; i < nx; i++) {



625          x1[i] += Delta[i];



626          ztmp[i] = (x1[i] - x1start[i])/hN;



627        }



628      }



629    }



630

632       E(:,j) = ytmp



633       for k = j:-1:2



634       coef = N(k-1)/(N(j) - N(k-1))



635       E(:,k-1) = E(:,k) + coef*( E(:,k) - E(:,k-1) )



636       end



637     */



638    (void) memcpy(&(E[nx*j]), x1,



639                  (uint_T)nx*sizeof(real_T));



640    for (k = j; k > 0; k--) {

644      }



645



646      (void) memcpy(&(E[nx*(k-1)]), x1,



647                    (uint_T)nx*sizeof(real_T));



648    }



649

653    }

656                  (uint_T)nx*sizeof(real_T));



657    for (k = j; k > 0; k--) {



658      real_T coef = (real_T)(N[k-1]) / (N[j]-N[k-1]);



659      for (i = 0; i < nx; i++) {



660        xdot[i] = Edot[nx*k+i] + coef*(Edot[nx*k+i] - Edot[nx*(k-1)+i]);



661      }



662



663      (void) memcpy(&(Edot[nx*(k-1)]), xdot,



664                    nx*sizeof(real_T));



665    }



666  }



667



668  /* x1 = E(:,1); */



669  (void) memcpy(x1, E,



670                (uint_T)nx*sizeof(real_T));



671



672  /* Extrapolated xdot */



673  (void) memcpy(xdot, Edot,



674                nx*sizeof(real_T));



675



676  /* t1 = t0 + h; */



677  rtsiSetT(si,rtsiGetSolverStopTime(si));



678  rtsiSetSimTimeStep(si,MAJOR_TIME_STEP);



679}
